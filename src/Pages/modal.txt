// App.js

import React, { useState } from 'react';
import Modal from './Modal'; // Make sure to provide the correct path

const App = () => {
  const [isModalOpen, setIsModalOpen] = useState(false);
  const [rentedMovieName, setRentedMovieName] = useState('');

  const handleRentMovie = (movieName) => {
    setRentedMovieName(movieName);
    setIsModalOpen(true);
  };

  const handleCloseModal = () => {
    setIsModalOpen(false);
    setRentedMovieName('');
  };

  return (
    <div>
      {/* Your movie listing and rental logic */}
      {/* When renting a movie, call handleRentMovie(movieName) */}
      
      {isModalOpen && (
        <Modal movieName={rentedMovieName} onClose={handleCloseModal} />
      )}
    </div>
  );
};

export default App;
